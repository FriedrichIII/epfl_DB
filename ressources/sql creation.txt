-- utiliser une table pour chaque type de médaille?, ex-aequo?
-- ou Event -> goldMedaillist, siverMedaillist, bronzeMedaillist 
-- + ex aequo 12 / ex aequo 23
CREATE TABLE Medals
( medalID INTEGER,
  type CHAR(8),
  athleteID INTEGER,
  eventID INTEGER,
  PRIMARY_KEY (medalID),
  FOREIGN_KEY (type) REFERENCES MedalTypes,
  FOREIGN_KEY (athleteID) REFERENCES Athletes.
  FOREIGN_KEY (eventID) REFERENCES Events
)

CREATE TABLE MedalTypes
( type CHAR(8),
  PRIMARY_KEY (type)
)

CREATE TABLE Events
( eventID INTEGER,
  name CHAR(128),
  gameID INTEGER,
  disciplineID INTEGER,
  PRIMARY_KEY (eventID),
  FOREIGN_KEY (gameID) REFERENCES Games,
  FOREIGN_KEY (disciplineID) REFERENCES Sports
)

CREATE TABLE Disciplines
( disciplineID INTEGER,
  name CHAR(128),
  sportID INTEGER,
  PRIMARY_KEY (disciplineID),
  FOREIGN_KEY (sportID) REFERENCES Sports
)

CREATE TABLE Sports
( sportID INTEGER,
  name CHAR(128),
  PRIMARY_KEY (sportID)
)
